14:     def escape(string)
15:       result = ((translation_to && translation_from) ? Iconv.iconv(translation_to, translation_from, string) : string).to_s
16:       result.gsub!(/[^\x00-\x7F]+/, '') # Remove anything non-ASCII entirely (e.g. diacritics).
17:       result.gsub!(/[^\w_ \-]+/i,   '') # Remove unwanted chars.
18:       result.gsub!(/[ \-]+/i,      '-') # No more than one of the separator in a row.
19:       result.gsub!(/^\-|\-$/i,      '') # Remove leading/trailing separator.
20:       result.downcase!
21:       result.size.zero? ? random_permalink(string) : result
22:     rescue
23:       random_permalink(string)
24:     end
